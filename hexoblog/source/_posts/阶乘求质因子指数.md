#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;

typedef long long ll;
const int N = 1e7 + 10;

int cnt;
bool vis[N];
int prime[N];

void init_prime(const int n) {
	for (int i = 2; i <= n; i++) {
		if (!vis[i]) {
			prime[++cnt] = i;
		}
		for (int j = 1; j <= cnt && i <= n / prime[j]; j++) {
			vis[i * prime[j]] = 1;
			if (i % prime[j] == 0) {
				break;
			}
		}
	}
}
ll get(int n, int p) {
	ll res = 0;
	while (n) {
		res += n / p;
		n /= p;
	}
	return res;
}
ll get(int l, int r, int p) {
	return get(r, p) - get(l - 1, p);
}
bool judge(int l1, int r1, int l2, int r2) {
	for (int i = 1; i <= cnt; i++) {
		int p = prime[i];
		int cnt1 = get(l1, r1, p);
		int cnt2 = get(l2, r2, p);

		if (cnt1 && !cnt2) return 0;
		if (cnt1 > cnt2) return 0;
	}
	return 1;
}


int main() {

	init_prime(1e7);

	int T; cin >> T;

	while (T--) {
		int l1, r1, l2, r2;
		cin >> l1 >> r1 >> l2 >> r2;

		if (judge(l1, r1, l2, r2)) cout << "Yes" << endl;
		else cout << "No" << endl;
	}
	return 0;
}
